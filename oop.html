Object Oriented Programming 

 
    /**
    From an assignment given about 
    object oriented programming
    */
    public class Airplane{ /** identity */  
        /** characteristics */
        private int gasLevel;
        private String landingGear;
        private String doorStatus;
        private int passengers;
        
        public Airplane(){
            gasLevel = 100;
            landingGear = "Down";
            doorStatus = "Open";
            passengers = 100;
        }   
        /** behaviours*/
        public void openDoor(){
            doorStatus = "Open";
        }
        public void closeDoor(){
            doorStatus = "Closed";
        }
    }
